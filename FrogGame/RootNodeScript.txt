extends CharacterBody2D

const JUMP_VELOCITY = -520.0
var can_attack: bool = true
var is_attacking: bool = false

@onready var animated_sprite = $AnimatedSprite2D
@onready var AttackRay = $AttackRay

func _ready() -> void:
	animated_sprite.animation_finished.connect(_on_animation_finished)
	
	# 1. Connect the frame_changed signal so we can monitor active attack frames
	animated_sprite.frame_changed.connect(_on_sprite_frame_changed)
	

func _physics_process(delta: float) -> void:
	if not is_on_floor():
		velocity += get_gravity() * delta

	if Input.is_action_just_pressed("ui_accept") and is_on_floor() and not is_attacking:
		velocity.y = JUMP_VELOCITY

	var direction := Input.get_axis("ui_left", "ui_right")
	
	# 1. The gatekeeper: If we are already in a cooldown, ignore all input below
	if can_attack:
		var pressed_left := Input.is_action_just_pressed("ui_left")
		var pressed_right := Input.is_action_just_pressed("ui_right")
		
		# 2. Check if either direction was just clicked once
		if pressed_left or pressed_right:
			can_attack = false # Instantly lock out further inputs
			is_attacking = true
			
			if pressed_left:
				animated_sprite.flip_h = true  
				AttackRay.scale.y = -1       
				print("left click")
			else:
				animated_sprite.flip_h = false 
				AttackRay.scale.y = 1        
				print("right click")
			
			animated_sprite.play("Attack")
			
			# 3. The Cooldown: Wait for the attack animation to finish playing completely
			await animated_sprite.animation_finished
			
			# 4. Reset states so the player can click to attack again
			is_attacking = false
			can_attack = true

	velocity.x = 0

	if not is_attacking:
		if not is_on_floor():
			animated_sprite.play("Hop")
		else:
			animated_sprite.play("Idle")

	move_and_slide()

# 2. This function runs automatically every single time a new frame displays
func _on_sprite_frame_changed() -> void:
	if animated_sprite.animation == "Attack":
		# CHANGE THESE NUMBERS: If your attack strikes on frame 2 and 3:
		if animated_sprite.frame == 2 or animated_sprite.frame == 3:
			AttackRay.enabled = true # Turn hitbox ON
		else:
			AttackRay.enabled = false  # Turn hitbox OFF for anticipation/recovery frames

func _on_animation_finished() -> void:
	if animated_sprite.animation == "Attack":
		is_attacking = false 
		AttackRay.enabled = false # Safety check to guarantee it's off when done
