extends CharacterBody2D

const JUMP_VELOCITY = -520.0

var is_attacking: bool = false

@onready var animated_sprite: AnimatedSprite2D = $AnimatedSprite2D
@onready var AttackBox = $AttackBox

func _ready() -> void:
	animated_sprite.animation_finished.connect(_on_animation_finished)
	
	# 1. Connect the frame_changed signal so we can monitor active attack frames
	animated_sprite.frame_changed.connect(_on_sprite_frame_changed)
	
	# Ensure the hitbox starts disabled
	AttackBox.disabled = true

func _physics_process(delta: float) -> void:
	if not is_on_floor():
		velocity += get_gravity() * delta

	if Input.is_action_just_pressed("ui_accept") and is_on_floor() and not is_attacking:
		velocity.y = JUMP_VELOCITY

	var direction := Input.get_axis("ui_left", "ui_right")
	
	if direction:
		if direction < 0:
			animated_sprite.flip_h = true  # Facing left
			# Flip the hitbox position horizontally to match where you face
			AttackBox.scale.x = -1
		else:
			animated_sprite.flip_h = false # Facing right
			AttackBox.scale.x = 1
		
		if not is_attacking:
			is_attacking = true
			animated_sprite.play("Attack")

	velocity.x = 0

	if not is_attacking:
		if not is_on_floor():
			animated_sprite.play("Hop")
		else:
			animated_sprite.play("Idle")

	move_and_slide()

# 2. This function runs automatically every single time a new frame displays
func _on_sprite_frame_changed() -> void:
	if animated_sprite.animation == "Attack":
		# CHANGE THESE NUMBERS: If your attack strikes on frame 2 and 3:
		if animated_sprite.frame == 2 or animated_sprite.frame == 3:
			AttackBox.disabled = false # Turn hitbox ON
		else:
			AttackBox.disabled = true  # Turn hitbox OFF for anticipation/recovery frames

func _on_animation_finished() -> void:
	if animated_sprite.animation == "Attack":
		is_attacking = false 
		AttackBox.disabled = true # Safety check to guarantee it's off when done
