extends CharacterBody2D

# Removed SPEED since the player shouldn't move horizontally
const JUMP_VELOCITY = -520.0

# Track if the player is currently locked in the attack animation
var is_attacking: bool = false

@onready var animated_sprite: AnimatedSprite2D = $AnimatedSprite2D
@onready var AttackBox = $AttackBox

func _ready() -> void:
	# Connect Godot's built-in signal to know when the attack finishes
	animated_sprite.animation_finished.connect(_on_animation_finished)

func _physics_process(delta: float) -> void:
	# Add the gravity so jumping works
	if not is_on_floor():
		velocity += get_gravity() * delta

	# Handle jump (Only if not attacking)
	if Input.is_action_just_pressed("ui_accept") and is_on_floor() and not is_attacking:
		velocity.y = JUMP_VELOCITY

	# Get the input direction
	var direction := Input.get_axis("ui_left", "ui_right")
	
	# 1. ALWAYS handle turning around instantly, even while attacking!
	if direction:
		if direction < 0:
			animated_sprite.flip_h = true  # Facing left
		else:
			animated_sprite.flip_h = false # Facing right
		
		# 2. Only start a NEW attack animation if we aren't already attacking
		if not is_attacking:
			is_attacking = true
			AttackBox.disabled = false
			animated_sprite.play("Attack")
			AttackBox.disabled = true

	# Ensure the player doesn't slide around horizontally
	velocity.x = 0

	# --- ANIMATION LOGIC ---
	# Only manage Hop/Idle if the player isn't busy attacking
	if not is_attacking:
		if not is_on_floor():
			animated_sprite.play("Hop")
		else:
			animated_sprite.play("Idle")

	move_and_slide()

# This automatically runs the moment the "Attack" animation ends
func _on_animation_finished() -> void:
	if animated_sprite.animation == "Attack":
		is_attacking = false # Release the lock so they go back to Idle/Hop
