extends CharacterBody2D

const SPEED = 300.0
# Increased slightly to give you roughly a 200px vertical leap under default gravity
const JUMP_VELOCITY = -520.0

@onready var animated_sprite: AnimatedSprite2D = $AnimatedSprite2D

func _physics_process(delta: float) -> void:
	# Add the gravity (Crucial so the jump actually pulls the player back down!)
	if not is_on_floor():
		velocity += get_gravity() * delta

	# Handle jump.
	if Input.is_action_just_pressed("ui_accept") and is_on_floor():
		velocity.y = JUMP_VELOCITY

	# Get the input direction and handle the movement/deceleration.
	var direction := Input.get_axis("ui_left", "ui_right")
	if direction:
		# Flip the sprite horizontally depending on the direction
		if direction < 0:
			animated_sprite.flip_h = true  # Moving left -> Flip it
		else:
			animated_sprite.flip_h = false # Moving right -> Normal

	# --- ANIMATION LOGIC ---
	# Check if the player is in the air or on the ground
	if not is_on_floor():
		animated_sprite.play("jump")
	else:
		animated_sprite.play("Idle")

	move_and_slide()
